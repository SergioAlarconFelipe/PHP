<?php
// --- salarcon: print_d --- //
include 'extra_debug.php';

/**
 * Funcion depuracion de variables.
 * @since   2020-01-24
 * @param   array       param           Semaforo de control para imprimir los CSS y JS necesarios en la depuracion de variables solo una vez.
 * @see     print_d_StylesScripts
 */
$printed_StylesScripts = false;

/**
 * Funcion depuracion de variables.
 * @since   2019-05-30
 * @param   array       param           Array de parametros sobre que mostrar de la variable:
 *  <ul>
 *      <li>'name'      => String en el que indicamos el nombre de la variable indicada</li>
 *      <li>'return'    => String en el que indicamos si se imprimira o se devolvera el resultado</li>
 *      <li>'title'     => String que sera el titulo de la variable</li>
 *      <li>'args'      => Boolean que indica si nostramos o no los argumentos de cada funcion llamada</li>
 *      <li>'colors'    => Boolean que indica si nostramos o no los colores de las variables</li>
 *  </ul>
 * @return  string      Si se ha indicado se devuelve los datos de la variable, si no es null.
 */
function print_d( $element, $opt = array() ) {
    global $printed_StylesScripts;
    
    $salida = '';
    if( !$printed_StylesScripts ) {
        $salida .= print_d_StylesScripts();
        $printed_StylesScripts = true;
    }
    
    $salida .= '<pre style="text-align: left;">';
    // title
    if( isset( $opt['title'] ) && !empty( $opt['title'] ) ) {
        $salida .= '<div style="font-weight: bold; font-size: large; text-decoration: underline;">' . $opt['title'] . '</div>';
    }
    
    // trace
    foreach( array_reverse( debug_backtrace() ) as $trace ) {
        if( ( !empty( $trace['file'] ) || !empty( $trace['line'] ) ) && $trace['function'] != 'array_walk' ) {
            $parameters = '';
            if( !isset( $opt['args'] ) || ( isset( $opt['args'] ) && $opt['args'] == true ) ) {
                $args = array();
                if( isset( $trace['args'][0] ) && is_array( $trace['args'][0] ) && $trace['function'] != __FUNCTION__ && $trace['function'] != 'prints_d' ) {
                    $args = $trace['args'][0];
                } elseif( isset( $trace['args'] ) && $trace['function'] != __FUNCTION__ && $trace['function'] != 'prints_d' ) {
                    $args = $trace['args'];
                }
                
                if( !empty( $args ) ) {
                    foreach( $args as $key => $arg ) {
                        $parameters .= $key . ': ' . json_encode( $arg, JSON_PRETTY_PRINT | JSON_UNESCAPED_UNICODE | JSON_UNESCAPED_SLASHES ) . PHP_EOL;
                    }
                }
            }
            
            //$salida .= '<div style="font-weight: bold;" title="' . str_replace( '"', '\'', $parameters ) . '">' . $trace['file'] . ': ' . $trace['line'];
            $salida .= '<div style="position: relative;"><span class="traza" style="font-weight: bold; cursor: pointer;" ondblclick="toggleCollapseArgs( this )">' . $trace['file'] . ': ' . $trace['line'];
            if( $trace['function'] != __FUNCTION__ && $trace['function'] != 'prints_d' ) {
                $salida .= ' => ' . $trace['function'];
            }
            if( !empty( $parameters ) ) {
                $salida .= ' <span class="collapsable">[+]</span>';
            }
            $salida .= '</span>';
            $salida .= '<div class="args">';
            if( !empty( $parameters ) ) {
                 $salida .= str_replace( '"', '\'', $parameters );
            }
            $salida .= '</div>';
            $salida .= '</div>';
            
        }
    }
    
    // type and length
    $typelen = '';
    $value = '';
    $color = '';
    switch( gettype( $element ) ) {
        case 'string':
            $typelen = '[' . gettype( $element ) . ' (' . strlen($element) . ')]';
            $value = print_r( $element, true );
            if( !isset( $opt['colors'] ) || ( isset( $opt['colors'] ) && $opt['colors'] == true ) ) {
                $color = 'e8b46a';
            }
            break;
        case 'integer':
        case 'double':
            $typelen = '[' . gettype( $element ) . ' (' . strlen($element) . ')]';
            $value = print_r( $element, true );
            if( !isset( $opt['colors'] ) || ( isset( $opt['colors'] ) && $opt['colors'] == true ) ) {
                $color = '4caf50';
            }
            break;
        case 'array':
            $typelen = '[array (' . sizeof( $element ) . ')]';
            $value = print_r( $element, true );
            if( !isset( $opt['colors'] ) || ( isset( $opt['colors'] ) && $opt['colors'] == true ) ) {
                $color = 'a681d6';
            }
            break;
        case 'object':
            $typelen = '[object (' . sizeof( (array) $element ) . ')]';
            $value = print_r( $element, true );
            if( !isset( $opt['colors'] ) || ( isset( $opt['colors'] ) && $opt['colors'] == true ) ) {
                $color = '15b8cf';
            }
            break;
        case 'boolean':
            $typelen = '[' . gettype( $element ) . ']';
            if( $element ) {
                $value = 'true';
            } else {
                $value = 'false';
            }
            if( !isset( $opt['colors'] ) || ( isset( $opt['colors'] ) && $opt['colors'] == true ) ) {
                $color = '795d27';
            }
            break;
        default:
            $typelen = '[' . gettype( $element ) . ']';
            $value = print_r( $element, true );
            if( !isset( $opt['colors'] ) || ( isset( $opt['colors'] ) && $opt['colors'] == true ) ) {
                $color = '848877';
            }
            break;
    }
    if( !empty( $color ) ) {
        $color = 'color: #' . $color . ';';
    }
    
    // values
    if(isset($opt['name']) && !empty($opt['name'])) {
        $salida .= '<div style="padding-left: 16px;"><u>' . $opt['name'] . '</u> ' . $typelen . ': </div><div style="padding-left: 32px; ' . $color . '">' . $value . '</div>';
    } else {
        $salida .= '<div style="padding-left: 16px;">' . $typelen . ': </div><div style="padding-left: 32px; ' . $color . '">' . $value . '</div>';
    }
    
    $salida .= '</pre>';
    
    // output
    if(isset($opt['return']) && $opt['return'] == true) {
        return $salida;
    } else {
        echo $salida;
        return null;
    }
}
/**
 * Funcion para la depuracion de variables en grupo.
 * @since   2019-06-10
 * @param   mixed $expression [, mixed $... ]       Una o mas variables a depurar
 * @return  null
 */
function prints_d() {
    array_walk( func_get_args(), 'print_d' );
}

/**
 * Funcion para la insercion de los CSS y JS necesarios en la depuracion de variables.
 * @since   2020-01-24
 * @return  string      CSS y JS a insertar.
 */
function print_d_StylesScripts() {
    $salida = '
        <style>
            div > span.traza + div.args {
                display: none;
                padding-bottom: 4px;
                position: absolute;
                background-color: white;
                z-index: 1;
                padding: 5px;
                width: 98%;
                border-left: 1px solid black;
                border-bottom: 1px solid black;
                border-right: 1px solid black;
                border-bottom-left-radius: 8px;
                border-bottom-right-radius: 8px;
            }
            div > span.traza.showArgs + div.args {
                display: block;
                /*
                -webkit-box-shadow: 10px 10px 5px 0px rgba(0,0,0,0.75);
                -moz-box-shadow: 10px 10px 5px 0px rgba(0,0,0,0.75);
                box-shadow: 10px 10px 5px 0px rgba(0,0,0,0.75);
                */
            }
            div > span.traza.showArgs {
                padding: 5px;
                display: inline-block;
                width: 98%;
                border-left: 1px solid black;
                border-top: 1px solid black;
                border-right: 1px solid black;
                border-top-left-radius: 8px;
                border-top-right-radius: 8px;
                /*
                -webkit-box-shadow: 10px 10px 5px 0px rgba(0,0,0,0.75);
                -moz-box-shadow: 10px 10px 5px 0px rgba(0,0,0,0.75);
                box-shadow: 10px 10px 5px 0px rgba(0,0,0,0.75);
                */
            }
        </style>';
    
    $salida .= '
        <script>
            function toggleCollapseArgs( element ) {
                if( !element.classList.contains( \'showArgs\' ) ) {
                    var elements = document.getElementsByClassName( \'showArgs\' );
                    while( elements.length ) {
                        if( elements.item( 0 ).firstElementChild !== null ) {
                            elements.item( 0 ).firstElementChild.innerText = \'[+]\';
                        }
                        elements.item( 0 ).classList.remove( \'showArgs\' );
                    }
                    
                    element.classList.add( \'showArgs\' );
                    if( element.firstElementChild !== null ) {
                        if( element.firstElementChild.classList.contains( \'collapsable\' ) ) {
                            element.firstElementChild.innerText = \'[-]\';
                        }
                    }
                } else {
                    element.classList.remove( \'showArgs\' );
                    if( element.firstElementChild !== null ) {
                        if( element.firstElementChild.classList.contains( \'collapsable\' ) ) {
                            element.firstElementChild.innerText = \'[+]\';
                        }
                    }
                }
            }
        </script>
    ';
    
    return $salida;
}
